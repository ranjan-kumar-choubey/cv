import matplotlib.pyplot as plt

# Function to calculate Euclidean distance between two points
def distance(p1, p2):
    return np.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

# Function to filter points within a given radius from a center point
def canonical_disk_set(center, radius, points):
    return [p for p in points if distance(center, p) <= radius]

# Function to find circumcenter of three points
def circumcenter(p1, p2, p3):
    x1, y1 = p1
    x2, y2 = p2
    x3, y3 = p3

    D = 2 * (x1 * (y2 - y3) + x2 * (y3 - y1) + x3 * (y1 - y2))

    if D == 0:
        return None, None

    Ux = ((x1**2 + y1**2) * (y2 - y3) + (x2**2 + y2**2) * (y3 - y1) + (x3**2 + y3**2) * (y1 - y2)) / D
    Uy = ((x1**2 + y1**2) * (x3 - x2) + (x2**2 + y2**2) * (x1 - x3) + (x3**2 + y3**2) * (x2 - x1)) / D

    return Ux, Uy

# Function to calculate circumradius given a center and a point
def circumradius(center, p):
    return np.sqrt((center[0] - p[0])**2 + (center[1] - p[1])**2)

# Function to plot points and disk sets in an enhanced visual style
def plot_points_and_disks(points, disk_sets):
    plt.figure(figsize=(8, 8))
    plt.scatter(*zip(*points), label='Points', color='black', s=40, edgecolors='purple', zorder=1)

    for ds in disk_sets:
        circle = plt.Circle(ds[0], ds[1], fill=False, color='purple', linestyle='-', alpha=1)
        plt.gca().add_patch(circle)

    plt.xlabel('X-axis')
    plt.ylabel('Y-axis')
    plt.title('Canonical Disk Sets', fontsize=16)
    plt.legend()

    plt.gca().spines['top'].set_visible(False)
    plt.gca().spines['right'].set_visible(False)

    plt.gca().set_aspect('equal', adjustable='datalim')
    plt.tight_layout()
    plt.show()

# Function to find all possible distinct canonical disk sets
def find_all_disk_sets(points):
    disk_sets = []
    
    # Loop to find canonical disks from pairs of points
    for i in range(len(points)):
        for j in range(i + 1, len(points)):
            center = ((points[i][0] + points[j][0]) / 2, (points[i][1] + points[j][1]) / 2)
            radius = distance(points[i], center)
            disk_sets.append((center, radius))
    
    # Loop to find canonical disks from sets of three points
    for i in range(len(points)):
        for j in range(i + 1, len(points)):
            for k in range(j + 1, len(points)):
                x, y = circumcenter(points[i], points[j], points[k])
                if x is not None and y is not None:
                    center = (x, y)
                    radius = circumradius([x, y], points[i])
                    disk_sets.append((center, radius))

    return disk_sets

import random
import matplotlib.pyplot as plt
import numpy as np

# Generate random points within the given range
def generate_points(num_points):
    points = []
    for _ in range(num_points):
        x = random.uniform(-20, 20)
        y = random.uniform(-20, 20)
        points.append((x, y))
    return points

# Calculate Euclidean distance between two points
def distance(p1, p2):
    return np.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

# Find the smallest disk containing the given points with its center on the x-axis
def find_smallest_disk(points):
    all_info_list = []
    for i in range(len(points)):
        for j in range(i + 1, len(points)):
            # Calculate midpoint between two points
            mid_x = (points[j][0] + points[i][0]) / 2
            mid_y = (points[j][1] + points[i][1]) / 2

            # Calculate distance from midpoint to all points
            list_perp_intersection = [distance(points[k], (mid_x, mid_y)) for k in range(len(points))]
            list_perp_intersection.sort()
            mid_intersection_point = list_perp_intersection[len(points) // 2]
            max_dist = -99

            # Find the point with maximum distance from the midpoint
            for point in points:
                temp = (point[0] - mid_intersection_point)**2 + point[1]**2
                dist = np.sqrt(temp)
                if dist > max_dist:
                    max_dist = dist
                    max_dist_point = (point, (mid_intersection_point, 0))

            info = (max_dist_point, max_dist)
            all_info_list.append(info)

    # Get the smallest disk by minimizing the distance
    result = min(all_info_list, key=lambda x: x[1])
    return result

# Plot the points and the smallest disk containing them
def plot_smallest_disk(points, result):
    x_points = [x[0] for x in points]
    y_points = [y[1] for y in points]

    # Determine plot limits for better visualization
    min_x = min([x[0] for x in points]) - 2 * result[1]
    max_x = max([x[0] for x in points]) + 2 * result[1]
    min_y = min([x[1] for x in points]) - 2 * result[1]
    max_y = max([x[1] for x in points]) + 2 * result[1]

    # Plot points and the smallest disk
    plt.figure(figsize=(8, 6))
    plt.xlim([min_x, max_x])
    plt.ylim([min_y, max_y])
    plt.scatter(x_points, y_points, color='green', label='Random Points',s=10)

    my_circle = plt.Circle(result[0][1], result[1], fill=False, edgecolor='purple', linestyle='-', linewidth=1)

    plt.plot(result[0][1][0], result[0][1][1], color='purple', marker='.', markersize=8, label='Center of Disk')
    plt.title("Smallest Disk Centered on X-Axis")
    plt.xlabel('X-axis')
    plt.ylabel('Y-axis')
    plt.legend()
    plt.grid(True)
    plt.show()

import matplotlib.pyplot as plt
import numpy as np
import warnings

# Function to take user input for constraints
def get_constraints():
    # Get the number of constraints from the user
    num_constraints = int(input("Enter the number of constraints: "))
    constraints = []
    
    # Gather coefficients for each constraint from the user
    for i in range(num_constraints):
        constraint = tuple(map(float, input(f"Enter coefficients for constraint {i+1} (aX + bY <= c): ").split()))
        constraints.append(constraint)
    return constraints

# Function to solve the Integer Linear Programming problem
def solve_integer_linear_programming(constraints, obj_eqn):
    x_range = np.arange(0, 101, 1)  # Integer range from 0 to 100
    y_range = np.arange(0, 101, 1)  # Integer range from 0 to 100

    valid_points = []

    # Brute force iteration through integer ranges
    for x in x_range:
        for y in y_range:
            # Check if the point satisfies all constraints
            valid = all((a * x + b * y <= c and x >= 0 and y >= 0) for a, b, c in constraints)
            if valid:
                valid_points.append((x, y))

    max_value = float('-inf')
    optimal_point = None

    # Find the optimal solution and its value
    for point in valid_points:
        value = obj_eqn[0] * point[0] + obj_eqn[1] * point[1]
        if value > max_value:
            max_value = value
            optimal_point = point

    return optimal_point, max_value, valid_points

# Function to plot the Integer Linear Programming solution
def plot_integer_linear_programming(constraints, optimal_point, max_value, valid_points):
    x = np.arange(0, 101, 1)  # Integer range from 0 to 100

    with warnings.catch_warnings():
        warnings.simplefilter("ignore")
        plt.figure(figsize=(8, 6))

        # Plot each constraint
        for constraint_idx, constraint in enumerate(constraints, start=1):
            y = (constraint[2] - constraint[0] * x) / constraint[1]
            plt.plot(x, y, label=f'Constraint {constraint_idx}: {constraint[0]}X + {constraint[1]}Y <= {constraint[2]}')

        # Plot the feasible region
        y_feasible = np.minimum.reduce([(constraint[2] - constraint[0] * x) / constraint[1] for constraint in constraints])
        plt.fill_between(x, 0, y_feasible, where=(y_feasible >= 0), alpha=0.2)

        # Plot feasible points and optimal point
        plt.scatter([point[0] for point in valid_points], [point[1] for point in valid_points], color='green', label='Feasible Points',s=15)
        plt.scatter(optimal_point[0], optimal_point[1], color='red', label='Optimal Point',s=15)
        plt.xlabel('X1')
        plt.ylabel('X2')
        plt.title('Integer Linear Programming (Maximization Problem)')
        plt.legend()
        plt.grid()
        plt.show()
