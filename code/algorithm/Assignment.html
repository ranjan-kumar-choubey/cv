import matplotlib.pyplot as plt
import math
import random
from itertools import combinations

def dist(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def generate_random_points(num_points, x_min, x_max, y_min, y_max):
    return [(random.randint(x_min, x_max), random.randint(y_min, y_max)) for _ in range(num_points)]

def find_distinct_canonical_disks(P):
    distinct_disks = []
    n = len(P)
    
    for i, j, k in combinations(range(n), 3):
        # Calculate the circumcenter and circumradius of the triangle formed by points i, j, and k
        xi, yi = P[i]
        xj, yj = P[j]
        xk, yk = P[k]
        
        d = 2 * (xi * (yj - yk) + xj * (yk - yi) + xk * (yi - yj))
        if d == 0:
            # Points are collinear, skip this combination
            continue
        
        ux = ((xi**2 + yi**2) * (yj - yk) + (xj**2 + yj**2) * (yk - yi) + (xk**2 + yk**2) * (yi - yj)) / d
        uy = ((xi**2 + yi**2) * (xk - xj) + (xj**2 + yj**2) * (xi - xk) + (xk**2 + yk**2) * (xj - xi)) / d
        
        circumcenter = (round(ux, 1), round(uy, 1))  # Round to 1 decimal place
        circumradius = round(dist(circumcenter, P[i]), 1)  # Round to 1 decimal place
        
        valid = True
        for p in P:
            if dist(circumcenter, p) > circumradius:
                valid = False
                break
        
        if valid:
            distinct_disks.append((circumcenter, circumradius))
    
    return distinct_disks

import itertools
import random
import sys
import timeit
import matplotlib.pyplot as plt

def BruteForce_Sorting(input_size):
    input_data = [random.randrange(1, 100) for i in range(input_size)]
    start_time = timeit.default_timer()

    for p in itertools.permutations(input_data):
        if all(p[i] <= p[i + 1] for i in range(len(p) - 1)):
            return timeit.default_timer() - start_time

    return float('inf')

if __name__ == "__main__":
    input_sizes = range(1, 12)  # Input sizes from 1 to 15
    execution_times = []

    for input_size in input_sizes:
        print(f"Running with input size  = {input_size}")
        execution_time = BruteForce_Sorting(input_size)
        execution_times.append(execution_time)

    # Plotting the results
    plt.plot(input_sizes, execution_times, marker='o')
    plt.xlabel('Input Size')
    plt.ylabel('Execution Time (seconds)')
    plt.title('Execution Time vs. Input Size')
    plt.grid(True)
    plt.show()
