#include<stdio.h>

int main()
{
    int n;
    scanf("%d",&n);
    int arr[n][n];

    //taking input
    for(int i =0;i<n;i++)
        for(int j=0;j<n;j++)
            scanf("%d",&arr[i][j]);


    // upper part
    for(int i =0;i<n;i++){

        for(int m = 0;m<n-i;m++) //for spacing
            printf("  ");

        for(int j=0;j<=i;j++){  //for accesing element 
            printf(" %d",arr[i-j][j]);
            printf("  ");}
        printf("\n");
    }

    //lower part
    for(int i = n-2;i>=0;i--){
        
        for(int m = 0 ;m<=n-i-1;m++) //for spacing
            printf("  ");

        for(int j=0;j<=i;j++){        //for accesing element 
            printf(" %d",arr[n-1-j][j+n-1-i]);
            printf("  ");}
        printf("\n");
    }
}
